<script>
    // 지도 모듈 불러오기
    common.importJs([
        'ui/naver-map',
        'module/geolocation'
    ]).done(function() {
        var map;
        map = new common.ui.NaverMap('#naver_map');

        // resizeend 이벤트가 발생시 지도사이즈를 재조절
        $(window).on('resizeend', function() {

            map.setSize({
                width: common.util.getWinWidth() - 20,
                height: 600
            });

        }).triggerHandler('resizeend');

        // 내위치 가져오기
        var geo = new common.module.Geolocation();
        geo.getCurrentPosition({
            timeout: 20000 // 제한시간
        }).done(function(position) {

            // 지도에 마커 추가
            map.addMarker({
                title: '내위치',
                x: position.coords.latitude,
                y: position.coords.longitude
            });

        }).fail(function(err) {
            alert(err.message);
        });
    });

</script>
